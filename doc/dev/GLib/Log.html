def set_log_domain(domain)
  level =
    FLAG_RECURSION |
    FLAG_FATAL |
    LEVEL_ERROR |
    LEVEL_CRITICAL |
    LEVEL_WARNING
  if $VERBOSE or $DEBUG
    level |=
      LEVEL_MESSAGE |
      LEVEL_INFO
  end
  if $DEBUG
    level |= LEVEL_DEBUG
  end
  GLib::Log.set_handler(domain, level)
end
