def init(*argv)
  class << self
    remove_method(:init)
    remove_method(:const_missing)
    remove_method(:method_missing)
  end
  Gdk.init if Gdk.respond_to?(:init)
  loader = Loader.new(self, argv)
  loader.load
end

def method_missing(name, *args, &block)
  init()
  if respond_to?(name)
    __send__(name, *args, &block)
  else
    super
  end
end
